 app.post("api/v1/customRequest", async (req, res) => {
      console.log("Received custom req form data:", req.body);
      try {
        const reqData = req.body;
        const result = await customRequestCollection.insertOne(reqData);
        res.status(200).json(result)
      }
      catch (err) {
        console.error("Error:", err);
        res.status(500).send("Server error");
      } 
    })
